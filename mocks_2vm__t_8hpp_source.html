232 bsl::expects(online_pps <= m_active.size());

234 for (bsl::safe_idx mut_i{}; mut_i < online_pps; ++mut_i) {
235 if (*m_active.at_if(mut_i)) {
236 return bsl::to_u16(mut_i);

239 bsl::touch();
