68 tlb_flush(
69 bsl::safe_u64 const &addr, bsl::safe_u16 const &asid = syscall::BF_INVALID_ID) noexcept
70 {
71 bsl::expects(addr.is_valid_and_checked());
72 bsl::expects(addr.is_pos());
73 bsl::expects(asid.is_valid_and_checked());
74
75 if (bsl::unlikely(asid.is_zero())) {
76 bsl::alert() << "unable to flush the tlb for an ASID of 0\n" << bsl::here();
77 return;
78 }
79
80 if (syscall::BF_INVALID_ID == asid) {
81 return intrinsic_invlpg(addr.get());
82 }
83
84 return intrinsic_invlpga(addr.get(), bsl::to_u64(asid).get());
85 }
