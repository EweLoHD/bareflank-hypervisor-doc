74 [[nodiscard]] constexpr auto

76 tls_t &mut_tls,

78 intrinsic_t &mut_intrinsic,
79 vp_pool_t const &vp_pool,

81 {
82 auto const vpid{get_allocated_vpid(mut_tls.ext_reg1, vp_pool)};

100 bsl::print<bsl::V>() << bsl::here();
101 return syscall::BF_STATUS_FAILURE_UNKNOWN;
102 }
103
104 mut_tls.ext_reg0 = bsl::merge_umx_with_u16(mut_tls.ext_reg0, vsid).get();
105 return syscall::BF_STATUS_SUCCESS;

474 [[nodiscard]] constexpr auto
475 syscall_bf_vs_op_set_active(
476 tls_t &mut_tls,
477 intrinsic_t &mut_intrinsic,
478 vm_pool_t &mut_vm_pool,
479 vp_pool_t &mut_vp_pool,
480 vs_pool_t &mut_vs_pool,
481 ext_pool_t &mut_ext_pool,
482 bool const advance_ip) noexcept -> syscall::bf_status_t
483 {
484 // ---------------------------------------------------------------------
485 // Gather Arguments
486 // ---------------------------------------------------------------------
487
488 auto const vmid{get_allocated_vmid(mut_tls.ext_reg1, mut_vm_pool)};
489 if (bsl::unlikely(vmid.is_invalid())) {
490 bsl::print<bsl::V>() << bsl::here();
491 return syscall::BF_STATUS_INVALID_INPUT_REG1;
492 }
493
494 auto const vpid{get_allocated_vpid(mut_tls.ext_reg2, mut_vp_pool)};
495 if (bsl::unlikely(vpid.is_invalid())) {
496 bsl::print<bsl::V>() << bsl::here();
497 return syscall::BF_STATUS_INVALID_INPUT_REG2;
498 }
499
500 auto const vsid{get_locally_assigned_vsid(mut_tls, mut_tls.ext_reg3, mut_vs_pool)};
501 if (bsl::unlikely(vsid.is_invalid())) {
502 bsl::print<bsl::V>() << bsl::here();
503 return syscall::BF_STATUS_INVALID_INPUT_REG3;
504 }
505
506 // ---------------------------------------------------------------------
507 // Validate Assignment
508 // ---------------------------------------------------------------------
509
510 bool const vp_assigned_to_vm{is_vp_assigned_to_vm(mut_vp_pool, vpid, vmid)};
511 if (bsl::unlikely(!vp_assigned_to_vm)) {
512 bsl::print<bsl::V>() << bsl::here();
513 return syscall::BF_STATUS_INVALID_INPUT_REG2;
514 }
515
516 bool const vs_assigned_to_vp{is_vs_assigned_to_vp(mut_vs_pool, vsid, vpid)};
517 if (bsl::unlikely(!vs_assigned_to_vp)) {
518 bsl::print<bsl::V>() << bsl::here();
519 return syscall::BF_STATUS_INVALID_INPUT_REG3;
520 }
521
522 // ---------------------------------------------------------------------
523 // Active Status
524 // ---------------------------------------------------------------------
525
526 bool const vp_active_on_another_pp{is_vp_active_on_another_pp(mut_tls, mut_vp_pool, vpid)};
527 if (bsl::unlikely(vp_active_on_another_pp)) {
528 bsl::print<bsl::V>() << bsl::here();
529 return syscall::BF_STATUS_INVALID_INPUT_REG2;
530 }
531
532 bool const vs_active_on_another_pp{is_vs_active_on_another_pp(mut_tls, mut_vs_pool, vsid)};
533 if (bsl::unlikely(vs_active_on_another_pp)) {
534 bsl::print<bsl::V>() << bsl::here();
535 return syscall::BF_STATUS_INVALID_INPUT_REG3;
536 }
537
538 // ---------------------------------------------------------------------
539 // IP Advancement
540 // ---------------------------------------------------------------------
541
542 if (advance_ip) {
543 mut_vs_pool.advance_ip(mut_tls, mut_intrinsic, vsid);
544 }
545 else {
546 bsl::touch();
547 }
548
549 // ---------------------------------------------------------------------
550 // Activate
551 // ---------------------------------------------------------------------
552
553 bool const vm_active_on_this_pp{mut_vm_pool.is_active_on_this_pp(mut_tls, vmid)};
554 if (!vm_active_on_this_pp) {
555 mut_vm_pool.set_inactive(mut_tls, bsl::to_u16(mut_tls.active_vmid));
556 mut_vm_pool.set_active(mut_tls, vmid);
557 }
558 else {
559 bsl::touch();
560 }
561
562 bool const vp_active_on_this_pp{mut_vp_pool.is_active_on_this_pp(mut_tls, vpid)};
563 if (!vp_active_on_this_pp) {
564 mut_vp_pool.set_inactive(mut_tls, bsl::to_u16(mut_tls.active_vpid));
565 mut_vp_pool.set_active(mut_tls, vpid);
566 }
567 else {
568 bsl::touch();
569 }
570
571 bool const vs_active_on_this_pp{mut_vs_pool.is_active_on_this_pp(mut_tls, vsid)};
572 if (!vs_active_on_this_pp) {
573 mut_vs_pool.set_inactive(mut_tls, mut_intrinsic, bsl::to_u16(mut_tls.active_vsid));
574 mut_vs_pool.set_active(mut_tls, mut_intrinsic, vsid);
575 }
576 else {
577 bsl::touch();
578 }
579
580 mut_ext_pool.signal_vm_active(mut_tls, mut_intrinsic, vmid);
581
582 // ---------------------------------------------------------------------
583 // Done
584 // ---------------------------------------------------------------------
585
586 return syscall::BF_STATUS_SUCCESS;
587 }

670 [[nodiscard]] constexpr auto
671 dispatch_syscall_bf_vs_op(
672 tls_t &mut_tls,
673 page_pool_t &mut_page_pool,
674 intrinsic_t &mut_intrinsic,
675 vm_pool_t &mut_vm_pool,
676 vp_pool_t &mut_vp_pool,
677 vs_pool_t &mut_vs_pool,
678 ext_pool_t &mut_ext_pool) noexcept -> syscall::bf_status_t
679 {
680 if (bsl::unlikely(!verify_handle_for_current_ext(mut_tls))) {
681 bsl::print<bsl::V>() << bsl::here();
682 return syscall::BF_STATUS_FAILURE_INVALID_HANDLE;
683 }
684
685 if (bsl::unlikely(!is_the_active_ext_the_vmexit_ext(mut_tls))) {
686 bsl::print<bsl::V>() << bsl::here();
687 return syscall::BF_STATUS_INVALID_PERM_DENIED;
688 }
689
690 switch (syscall::bf_syscall_index(mut_tls.ext_syscall).get()) {
691 case syscall::BF_VS_OP_CREATE_VS_IDX_VAL.get(): {
692 auto const ret{syscall_bf_vs_op_create_vs(
693 mut_tls, mut_page_pool, mut_intrinsic, mut_vp_pool, mut_vs_pool)};
694 if (bsl::unlikely(ret != syscall::BF_STATUS_SUCCESS)) {
695 bsl::print<bsl::V>() << bsl::here();
696 return ret;
697 }
698
699 return ret;
700 }
701
702 case syscall::BF_VS_OP_DESTROY_VS_IDX_VAL.get(): {
703 auto const ret{syscall_bf_vs_op_destroy_vs(mut_tls, mut_page_pool, mut_vs_pool)};
704 if (bsl::unlikely(ret != syscall::BF_STATUS_SUCCESS)) {
705 bsl::print<bsl::V>() << bsl::here();
706 return ret;
707 }
708
709 return ret;
710 }
711
712 case syscall::BF_VS_OP_INIT_AS_ROOT_IDX_VAL.get(): {
713 auto const ret{syscall_bf_vs_op_init_as_root(mut_tls, mut_intrinsic, mut_vs_pool)};
714 if (bsl::unlikely(ret != syscall::BF_STATUS_SUCCESS)) {
715 bsl::print<bsl::V>() << bsl::here();
716 return ret;
717 }
718
719 return ret;
720 }
721
722 case syscall::BF_VS_OP_READ_IDX_VAL.get(): {
723 auto const ret{syscall_bf_vs_op_read(mut_tls, mut_intrinsic, mut_vs_pool)};
724 if (bsl::unlikely(ret != syscall::BF_STATUS_SUCCESS)) {
725 bsl::print<bsl::V>() << bsl::here();
726 return ret;
727 }
728
729 return ret;
730 }
731
732 case syscall::BF_VS_OP_WRITE_IDX_VAL.get(): {
733 auto const ret{syscall_bf_vs_op_write(mut_tls, mut_intrinsic, mut_vs_pool)};
734 if (bsl::unlikely(ret != syscall::BF_STATUS_SUCCESS)) {
735 bsl::print<bsl::V>() << bsl::here();

737 }
738
739 return ret;

741
742 case syscall::BF_VS_OP_RUN_IDX_VAL.get(): {
743 auto const ret{syscall_bf_vs_op_run(
744 mut_tls,
745 mut_intrinsic,
746 mut_vm_pool,
747 mut_vp_pool,
748 mut_vs_pool,
749 mut_ext_pool,
750 false)};
751 if (bsl::unlikely(ret != syscall::BF_STATUS_SUCCESS)) {
752 bsl::print<bsl::V>() << bsl::here();
753 return ret;
754 }
755
756 return ret;
757 }
758
759 case syscall::BF_VS_OP_RUN_CURRENT_IDX_VAL.get(): {
760 auto const ret{
761 syscall_bf_vs_op_run_current(mut_tls, mut_intrinsic, mut_vs_pool, false)};
762 if (bsl::unlikely(ret != syscall::BF_STATUS_SUCCESS)) {
763 bsl::print<bsl::V>() << bsl::here();
764 return ret;
765 }
766
767 return ret;
768 }
769
770 case syscall::BF_VS_OP_ADVANCE_IP_AND_RUN_IDX_VAL.get(): {
771 auto const ret{syscall_bf_vs_op_advance_ip_and_run(
772 mut_tls, mut_intrinsic, mut_vm_pool, mut_vp_pool, mut_vs_pool, mut_ext_pool)};
773 if (bsl::unlikely(ret != syscall::BF_STATUS_SUCCESS)) {
774 bsl::print<bsl::V>() << bsl::here();
775 return ret;
776 }
777
778 return ret;
779 }
780
781 case syscall::BF_VS_OP_ADVANCE_IP_AND_RUN_CURRENT_IDX_VAL.get(): {
782 auto const ret{syscall_bf_vs_op_advance_ip_and_run_current(
