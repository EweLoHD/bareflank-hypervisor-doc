128 [[nodiscard]] constexpr auto
129 allocate(
130 tls_t const &tls,
131 page_pool_t const &page_pool,
132 intrinsic_t const &intrinsic,
133 bsl::safe_u16 const &vmid,
134 bsl::safe_u16 const &vpid,
135 bsl::safe_u16 const &ppid) noexcept -> bsl::safe_u16
136 {
137 for (auto &mut_vs : m_pool) {

141
142 bsl::touch();

328 [[nodiscard]] constexpr auto
329 vs_assigned_to_vm(bsl::safe_u16 const &vmid) const noexcept -> bsl::safe_u16
330 {
331 bsl::expects(vmid.is_valid_and_checked());
332 bsl::expects(vmid != syscall::BF_INVALID_ID);
333
334 for (auto const &vs : m_pool) {
335 if (vs.assigned_vm() == vmid) {
336 return vs.id();
337 }
338
339 bsl::touch();
340 }
341
342 return bsl::safe_u16::failure();
343 }
