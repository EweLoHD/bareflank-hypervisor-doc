135 [[maybe_unused]] constexpr auto
136 allocate(
137 gs_t const &gs,
138 tls_t const &tls,
139 bf_syscall_t &mut_sys,
140 intrinsic_t const &intrinsic,
141 bsl::safe_u16 const &vpid,
142 bsl::safe_u16 const &ppid) noexcept -> bsl::safe_u16
143 {
144 auto const vsid{this->id()};
145
146 bsl::expects(vsid != BF_INVALID_ID);
147 bsl::expects(allocated_status_t::deallocated == m_allocated);
148
149 bsl::expects(vpid.is_valid_and_checked());
150 bsl::expects(vpid != BF_INVALID_ID);
151 bsl::expects(ppid.is_valid_and_checked());
152 bsl::expects(ppid != BF_INVALID_ID);
153
154 bsl::discard(gs);
155 bsl::discard(tls);
156 bsl::discard(intrinsic);
157
158 constexpr auto guest_asid_val{0x1_u64};
159 constexpr auto guest_asid_idx{bf_reg_t::bf_reg_t_guest_asid};
160 bsl::expects(mut_sys.bf_vs_op_write(this->id(), guest_asid_idx, guest_asid_val));
161
162 constexpr auto intercept1_val{0x00040000_u64};
163 constexpr auto intercept1_idx{bf_reg_t::bf_reg_t_intercept_instruction1};
164 bsl::expects(mut_sys.bf_vs_op_write(this->id(), intercept1_idx, intercept1_val));
165
166 constexpr auto intercept2_val{0x00000001_u64};
167 constexpr auto intercept2_idx{bf_reg_t::bf_reg_t_intercept_instruction2};
168 bsl::expects(mut_sys.bf_vs_op_write(this->id(), intercept2_idx, intercept2_val));
169
170 if (mut_sys.is_vs_a_root_vs(this->id())) {
171 bsl::expects(mut_sys.bf_vs_op_init_as_root(this->id()));
172 }
173 else {
174 bsl::touch();
175 }
176
177 m_assigned_vpid = ~vpid;
178 m_assigned_ppid = ~ppid;
179 m_allocated = allocated_status_t::allocated;
180
181 return vsid;
182 }
