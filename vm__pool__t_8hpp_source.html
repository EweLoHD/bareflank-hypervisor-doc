125 [[nodiscard]] constexpr auto
126 allocate(tls_t &mut_tls, page_pool_t &mut_page_pool, ext_pool_t &mut_ext_pool) noexcept
127 -> bsl::safe_u16
128 {
129 lock_guard_t mut_lock{mut_tls, m_lock};
130
131 for (auto &mut_vm : m_pool) {
132 if (mut_vm.is_deallocated()) {
133 return mut_vm.allocate(mut_tls, mut_page_pool, mut_ext_pool);
134 }
135
136 bsl::touch();
137 }
138
139 bsl::error() << "vm_pool_t out of vms\n" << bsl::here();
140 return bsl::safe_u16::failure();
