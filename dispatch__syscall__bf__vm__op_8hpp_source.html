246 [[nodiscard]] constexpr auto
247 dispatch_syscall_bf_vm_op(
248 tls_t &mut_tls,
249 page_pool_t &mut_page_pool,
250 intrinsic_t &mut_intrinsic,
251 vm_pool_t &mut_vm_pool,
252 vp_pool_t const &vp_pool,
253 vs_pool_t &mut_vs_pool,
254 ext_pool_t &mut_ext_pool) noexcept -> syscall::bf_status_t
255 {
256 if (bsl::unlikely(!verify_handle_for_current_ext(mut_tls))) {
257 bsl::print<bsl::V>() << bsl::here();
258 return syscall::BF_STATUS_FAILURE_INVALID_HANDLE;
259 }
260
261 if (bsl::unlikely(!is_the_active_ext_the_vmexit_ext(mut_tls))) {
262 bsl::print<bsl::V>() << bsl::here();
263 return syscall::BF_STATUS_INVALID_PERM_DENIED;
264 }
265
266 switch (syscall::bf_syscall_index(mut_tls.ext_syscall).get()) {
267 case syscall::BF_VM_OP_CREATE_VM_IDX_VAL.get(): {
268 auto const ret{
269 syscall_bf_vm_op_create_vm(mut_tls, mut_page_pool, mut_vm_pool, mut_ext_pool)};
270 if (bsl::unlikely(ret != syscall::BF_STATUS_SUCCESS)) {
271 bsl::print<bsl::V>() << bsl::here();
272 return ret;
273 }
274
275 return ret;
276 }
277
278 case syscall::BF_VM_OP_DESTROY_VM_IDX_VAL.get(): {
279 auto const ret{syscall_bf_vm_op_destroy_vm(
280 mut_tls, mut_page_pool, mut_vm_pool, vp_pool, mut_ext_pool)};
281 if (bsl::unlikely(ret != syscall::BF_STATUS_SUCCESS)) {
282 bsl::print<bsl::V>() << bsl::here();
283 return ret;
284 }
285
286 return ret;
287 }
288
289 case syscall::BF_VM_OP_MAP_DIRECT_IDX_VAL.get(): {
290 auto const ret{syscall_bf_vm_op_map_direct(mut_tls, mut_page_pool, mut_vm_pool)};
291 if (bsl::unlikely(ret != syscall::BF_STATUS_SUCCESS)) {
292 bsl::print<bsl::V>() << bsl::here();
293 return ret;
294 }
295
296 return ret;
297 }
298
299 case syscall::BF_VM_OP_UNMAP_DIRECT_IDX_VAL.get(): {
300 auto const ret{syscall_bf_vm_op_unmap_direct(
301 mut_tls, mut_page_pool, mut_intrinsic, mut_vm_pool)};
302 if (bsl::unlikely(ret != syscall::BF_STATUS_SUCCESS)) {
303 bsl::print<bsl::V>() << bsl::here();
304 return ret;
305 }
306
307 return ret;
308 }
309
310 case syscall::BF_VM_OP_UNMAP_DIRECT_BROADCAST_IDX_VAL.get(): {
311 auto const ret{syscall_bf_vm_op_unmap_direct_broadcast()};
312 if (bsl::unlikely(ret != syscall::BF_STATUS_SUCCESS)) {
313 bsl::print<bsl::V>() << bsl::here();
314 return ret;
315 }
316
317 return ret;
318 }
319
320 case syscall::BF_VM_OP_TLB_FLUSH_IDX_VAL.get(): {
321 auto const ret{
322 syscall_bf_vm_op_tlb_flush(mut_tls, mut_intrinsic, mut_vm_pool, mut_vs_pool)};
323 if (bsl::unlikely(ret != syscall::BF_STATUS_SUCCESS)) {
324 bsl::print<bsl::V>() << bsl::here();
325 return ret;
326 }
327
328 return ret;
329 }
330
331 default: {
332 break;
333 }
334 }
335
336 return report_syscall_unknown_unsupported(mut_tls);
337 }
