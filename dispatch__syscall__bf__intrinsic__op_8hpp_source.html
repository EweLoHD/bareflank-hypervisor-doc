101 [[nodiscard]] constexpr auto
102 dispatch_syscall_bf_intrinsic_op(tls_t &mut_tls, intrinsic_t &mut_intrinsic) noexcept
103 -> syscall::bf_status_t
104 {
105 if (bsl::unlikely(!verify_handle_for_current_ext(mut_tls))) {
106 bsl::print<bsl::V>() << bsl::here();
107 return syscall::BF_STATUS_FAILURE_INVALID_HANDLE;
108 }
109
110 switch (syscall::bf_syscall_index(mut_tls.ext_syscall).get()) {
111 case syscall::BF_INTRINSIC_OP_RDMSR_IDX_VAL.get(): {
112 auto const ret{syscall_bf_intrinsic_op_rdmsr(mut_tls, mut_intrinsic)};
113 if (bsl::unlikely(ret != syscall::BF_STATUS_SUCCESS)) {
114 bsl::print<bsl::V>() << bsl::here();
115 return ret;
116 }
117
118 return ret;
119 }
120
121 case syscall::BF_INTRINSIC_OP_WRMSR_IDX_VAL.get(): {
122 auto const ret{syscall_bf_intrinsic_op_wrmsr(mut_tls, mut_intrinsic)};
123 if (bsl::unlikely(ret != syscall::BF_STATUS_SUCCESS)) {
124 bsl::print<bsl::V>() << bsl::here();
125 return ret;
126 }
127
128 return ret;
129 }
130
131 default: {
132 break;
133 }
134 }
135
136 return report_syscall_unknown_unsupported(mut_tls);
137 }
