165 [[nodiscard]] static constexpr auto
166 size_to_page_aligned_bytes(bsl::safe_umx const &size) noexcept -> page_aligned_bytes_t
167 {
168 bsl::safe_umx mut_pages{};
169 if ((size % HYPERVISOR_PAGE_SIZE).checked().is_zero()) {
170 mut_pages = size >> HYPERVISOR_PAGE_SHIFT;
171 }
172 else {
173 mut_pages = (size >> HYPERVISOR_PAGE_SHIFT) + bsl::safe_umx::magic_1();
174 }
175
183
184 return {mut_pages * HYPERVISOR_PAGE_SIZE, mut_pages.checked()};
185 }

214 static constexpr void
215 validate_pt_load(bfelf::elf64_phdr_t const *const phdr) noexcept
216 {
217 bsl::expects(nullptr != phdr);
218
219 constexpr auto min_vaddr{HYPERVISOR_EXT_CODE_ADDR};
220 constexpr auto max_vaddr{(min_vaddr + HYPERVISOR_EXT_CODE_SIZE).checked()};
221
222 bsl::expects((phdr->p_vaddr) >= min_vaddr);
223 bsl::expects((phdr->p_vaddr + bsl::to_umx(phdr->p_memsz)).checked() <= max_vaddr);
224
225 if (bsl::safe_u32::magic_0() != (phdr->p_flags & bfelf::PF_W)) {
226 bsl::expects(bsl::safe_u32::magic_0() == (phdr->p_flags & bfelf::PF_X));
227 }
228 else {
229 bsl::touch();
230 }
231
232 if (bsl::safe_u32::magic_0() != (phdr->p_flags & bfelf::PF_X)) {
233 bsl::expects(bsl::safe_u32::magic_0() == (phdr->p_flags & bfelf::PF_W));
234 }
235 else {
236 bsl::touch();
237 }
238
239 bsl::expects(phdr->p_align == HYPERVISOR_PAGE_SIZE);
240 }

275 static constexpr void
276 validate(loader::ext_elf_file_t const *const file) noexcept
277 {
286
287 bsl::expects(nullptr != file);
288 validate_elf64_ehdr(file);
289
290 auto const phdrtab{get_phdrtab(file)};
291 for (bsl::safe_idx mut_i{}; mut_i < phdrtab.size(); ++mut_i) {
292 auto const *const phdr{phdrtab.at_if(mut_i)};
293
294 switch (phdr->p_type) {
295 case bfelf::PT_LOAD.get(): {
296 validate_pt_load(phdr);
297 break;
298 }
299
300 case bfelf::PT_GNU_STACK.get(): {
301 validate_pt_gnu_stack(phdr);
302 break;
303 }
304
305 case bfelf::PT_TLS.get(): {
306 validate_pt_tls(phdr);
307 break;
308 }
309
310 default: {
311 break;
312 }
313 }
314 }

331 allocate_page_for_add_segment(

333 page_pool_t &mut_page_pool,
334 root_page_table_t &mut_rpt,

347 if ((phdr->p_flags & bfelf::PF_X).is_pos()) {

371 [[nodiscard]] static constexpr auto
372 add_segment(
373 tls_t &mut_tls,
374 page_pool_t &mut_page_pool,
375 root_page_table_t &mut_rpt,
376 bfelf::elf64_phdr_t const *const phdr) noexcept -> bsl::errc_type
377 {
378 constexpr auto inc{bsl::to_idx(HYPERVISOR_PAGE_SIZE)};
379 bsl::span const segment{phdr->p_offset, bsl::to_umx(phdr->p_filesz)};
380
381 for (bsl::safe_idx mut_i{}; mut_i < phdr->p_memsz; mut_i += inc) {
382
383 auto *const pmut_page{
384 allocate_page_for_add_segment(mut_tls, mut_page_pool, mut_rpt, phdr, mut_i)};
385
386 if (bsl::unlikely(nullptr == pmut_page)) {
387 bsl::print<bsl::V>() << bsl::here();
388 return bsl::errc_failure;
389 }
390
400
401 auto const src{segment.subspan(mut_i, HYPERVISOR_PAGE_SIZE)};
402 if (src.empty()) {
403 continue;
404 }
405
406 bsl::builtin_memcpy(pmut_page->data.data(), src.data(), src.size());
407 }
408
409 return bsl::errc_success;
410 }

424 [[nodiscard]] static constexpr auto
425 add_segments(
426 tls_t &mut_tls,
427 page_pool_t &mut_page_pool,
428 root_page_table_t &mut_rpt,
429 loader::ext_elf_file_t const *const file) noexcept -> bsl::errc_type
430 {
431 auto mut_tls_vaddr{bsl::safe_u64::max_value()};
432
433 auto const phdrtab{get_phdrtab(file)};
434 for (bsl::safe_idx mut_i{}; mut_i < phdrtab.size(); ++mut_i) {
435 auto const *const phdr{phdrtab.at_if(mut_i)};
436
437 if (bfelf::PT_TLS == phdr->p_type) {
438 mut_tls_vaddr = bsl::to_u64(phdr->p_vaddr);
439 break;
440 }
441
442 bsl::touch();
443 }
444
445 for (bsl::safe_idx mut_i{}; mut_i < phdrtab.size(); ++mut_i) {
446 auto const *const phdr{phdrtab.at_if(mut_i)};
447
448 if (bfelf::PT_LOAD != phdr->p_type) {
449 continue;
450 }
451
457
458 if (phdr->p_vaddr == mut_tls_vaddr) {
459 continue;
460 }
461
462 auto const ret{add_segment(mut_tls, mut_page_pool, mut_rpt, phdr)};
463 if (bsl::unlikely(!ret)) {
464 bsl::print<bsl::V>() << bsl::here();
465 return bsl::errc_failure;
466 }
467
468 bsl::touch();
469 }
470
471 return bsl::errc_success;

673 [[nodiscard]] static constexpr auto
674 add_tls(
675 tls_t &mut_tls,
676 page_pool_t &mut_page_pool,
677 root_page_table_t &mut_rpt,
678 bsl::safe_u64 const &addr,
679 bfelf::elf64_phdr_t const *const phdr) noexcept -> bsl::errc_type
680 {
681 auto *const pmut_page{
682 mut_rpt.allocate_page<page_4k_t>(mut_tls, mut_page_pool, addr, MAP_PAGE_RW)};
683
684 if (bsl::unlikely(nullptr == pmut_page)) {
685 bsl::print<bsl::V>() << bsl::here();
686 return bsl::errc_failure;
687 }
688
689 if (bsl::safe_u64::magic_0() == phdr->p_filesz) {
690 return bsl::errc_success;
691 }
692
693 if (bsl::safe_u64::magic_0() == phdr->p_memsz) {
694 return bsl::errc_success;
695 }
696
702
703 bsl::span const src{phdr->p_offset, bsl::to_umx(phdr->p_filesz)};
704
711
712 auto const dst_idx{bsl::to_idx((HYPERVISOR_PAGE_SIZE - phdr->p_memsz).checked())};
713 bsl::builtin_memcpy(pmut_page->data.at_if(dst_idx), src.data(), src.size());
714
715 return bsl::errc_success;
716 }

851 [[nodiscard]] constexpr auto
852 initialize_rpt(
853 tls_t &mut_tls,
854 page_pool_t &mut_page_pool,
855 root_page_table_t &mut_rpt,
856 root_page_table_t const &system_rpt,
857 loader::ext_elf_file_t const *const file) noexcept -> bsl::errc_type
858 {
859 bsl::errc_type mut_ret{};
860
861 mut_ret = mut_rpt.initialize(mut_tls, mut_page_pool);
862 if (bsl::unlikely(!mut_ret)) {
863 bsl::print<bsl::V>() << bsl::here();
864 return bsl::errc_failure;
865 }
866
867 bsl::finally mut_release_on_error{
868 [&mut_tls, &mut_rpt, &mut_page_pool]() noexcept -> void {
869 mut_rpt.release(mut_tls, mut_page_pool);
870 }};
871
872 mut_rpt.add_tables(mut_tls, system_rpt);
873
874 mut_ret = this->add_segments(mut_tls, mut_page_pool, mut_rpt, file);
875 if (bsl::unlikely(!mut_ret)) {
876 bsl::print<bsl::V>() << bsl::here();
877 return bsl::errc_failure;
878 }
879
880 mut_ret = this->add_stacks(mut_tls, mut_page_pool, mut_rpt);
881 if (bsl::unlikely(!mut_ret)) {
882 bsl::print<bsl::V>() << bsl::here();
883 return bsl::errc_failure;
884 }
885
886 mut_ret = this->add_fail_stacks(mut_tls, mut_page_pool, mut_rpt);
887 if (bsl::unlikely(!mut_ret)) {
888 bsl::print<bsl::V>() << bsl::here();
889 return bsl::errc_failure;
890 }
891
892 mut_ret = this->add_tls_blocks(mut_tls, mut_page_pool, mut_rpt, file);
893 if (bsl::unlikely(!mut_ret)) {
894 bsl::print<bsl::V>() << bsl::here();
895 return bsl::errc_failure;
896 }
897
898 mut_release_on_error.ignore();
899 return bsl::errc_success;
900 }

965 [[nodiscard]] constexpr auto
966 execute(
967 tls_t &mut_tls,
968 intrinsic_t &mut_intrinsic,
969 bsl::safe_u64 const &ip,
970 bsl::safe_u64 const &arg0 = {},
971 bsl::safe_u64 const &arg1 = {}) noexcept -> bsl::errc_type
972 {
973 bsl::expects(ip.is_valid_and_checked());
974 bsl::expects(ip.is_pos());
975 bsl::expects(arg0.is_valid_and_checked());
976 bsl::expects(arg1.is_valid_and_checked());
977
978 auto *const pmut_rpt{m_direct_map_rpts.at_if(bsl::to_idx(mut_tls.active_vmid))};
979 bsl::expects(nullptr != pmut_rpt);
980
981 if (pmut_rpt->is_inactive(mut_tls)) {
982 pmut_rpt->activate(mut_tls, mut_intrinsic);
983 }
984 else {
985 bsl::touch();
986 }
987
988 if (mut_tls.ext != this) {
989 mut_tls.ext = this;
990 mut_tls.active_extid = this->id().get();
991 }
992 else {
993 bsl::touch();
994 }
995
996 if (ip == m_fail_ip) {
997 return call_ext(ip.get(), mut_tls.ext_fail_sp, arg0.get(), arg1.get());
998 }
999
1000 return call_ext(ip.get(), mut_tls.sp, arg0.get(), arg1.get());
1001 }

1016 [[nodiscard]] constexpr auto
1017 initialize(
1018 tls_t &mut_tls,
1019 page_pool_t &mut_page_pool,
1020 bsl::safe_u16 const &i,
1021 loader::ext_elf_file_t const *const file,
1022 root_page_table_t const &system_rpt) noexcept -> bsl::errc_type
1023 {
1024 bsl::expects(i.is_valid_and_checked());
1025 bsl::expects(i != syscall::BF_INVALID_ID);
1026 bsl::expects(nullptr != file);
1027
1028 validate(file);
1029 m_entry_ip = file->e_entry;
1030
1031 auto const ret{
1032 this->initialize_rpt(mut_tls, mut_page_pool, m_main_rpt, system_rpt, file)};
1033
1034 if (bsl::unlikely(!ret)) {
1035 bsl::print<bsl::V>() << bsl::here();
1036 return ret;
1037 }
1038
1039 m_id = ~i;
1040 m_handle = syscall::BF_INVALID_HANDLE;
1041 return ret;

1317 [[nodiscard]] constexpr auto
1318 alloc_huge(
1319 tls_t &mut_tls,
1320 page_pool_t &mut_page_pool,
1321 huge_pool_t &mut_huge_pool,
1322 bsl::safe_umx const &size) noexcept -> alloc_huge_t
1323 {
1324 bsl::expects(size.is_valid_and_checked());
1325 bsl::expects(size.is_pos());
1326
1327 if (bsl::unlikely(m_huge_allocs_idx >= HYPERVISOR_MAX_HUGE_ALLOCS)) {
1328 bsl::error() << "ext out of huge allocation slots\n" << bsl::endl;
1329 return {bsl::safe_u64::failure(), bsl::safe_u64::failure()};
1330 }
1331
1332 auto [mut_bytes, mut_pages]{size_to_page_aligned_bytes(size)};
1333 if (bsl::unlikely(mut_bytes.is_poisoned())) {
1334 bsl::print<bsl::V>() << bsl::here();
1335 return {bsl::safe_u64::failure(), bsl::safe_u64::failure()};
1336 }
1337
1338 auto mut_huge{mut_huge_pool.allocate(mut_tls, mut_pages)};
1339 if (bsl::unlikely(mut_huge.is_invalid())) {
1340 bsl::print<bsl::V>() << bsl::here();
1341 return {bsl::safe_u64::failure(), bsl::safe_u64::failure()};
1342 }
1343
1358
1359 *m_huge_allocs.at_if(m_huge_allocs_idx) = mut_huge;
1360 ++m_huge_allocs_idx;
1361
1362 auto const huge_phys{mut_huge_pool.virt_to_phys(mut_huge.data())};
1363 bsl::expects(huge_phys.is_valid_and_checked());
1364 bsl::expects(huge_phys.is_pos());
1365
1366 auto const huge_virt{(HYPERVISOR_EXT_HUGE_POOL_ADDR + huge_phys).checked()};
1367 bsl::expects(huge_virt.is_valid_and_checked());
1368 bsl::expects(huge_virt.is_pos());
1369
1377
1378 constexpr auto inc{bsl::to_idx(HYPERVISOR_PAGE_SIZE)};
1379 for (bsl::safe_idx mut_i{}; mut_i < mut_bytes; mut_i += inc) {
1380 auto const page_virt{(huge_virt + bsl::to_u64(mut_i)).checked()};
1381 auto const page_phys{(huge_phys + bsl::to_u64(mut_i)).checked()};
1382
1383 auto const ret{m_main_rpt.map(
1384 mut_tls, mut_page_pool, page_virt, page_phys, MAP_PAGE_RW, true)};
1385

1387 bsl::print<bsl::V>() << bsl::here();
1388 return {bsl::safe_u64::failure(), bsl::safe_u64::failure()};
1389 }
1390
1391 bsl::touch();
1392 }
1393
1394 this->update_direct_map_rpts(mut_tls);
1395 return {huge_virt, huge_phys};
1396 }
