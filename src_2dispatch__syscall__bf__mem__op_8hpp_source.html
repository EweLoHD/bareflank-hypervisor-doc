110 [[nodiscard]] constexpr auto
111 dispatch_syscall_bf_mem_op(
112 tls_t &mut_tls, page_pool_t &mut_page_pool, huge_pool_t &mut_huge_pool) noexcept
113 -> syscall::bf_status_t
114 {
115 if (bsl::unlikely(!verify_handle_for_current_ext(mut_tls))) {
116 bsl::print<bsl::V>() << bsl::here();
117 return syscall::BF_STATUS_FAILURE_INVALID_HANDLE;
118 }
119
120 switch (syscall::bf_syscall_index(mut_tls.ext_syscall).get()) {
121 case syscall::BF_MEM_OP_ALLOC_PAGE_IDX_VAL.get(): {
122 auto const ret{syscall_bf_mem_op_alloc_page(mut_tls, mut_page_pool)};
123 if (bsl::unlikely(ret != syscall::BF_STATUS_SUCCESS)) {
124 bsl::print<bsl::V>() << bsl::here();
125 return ret;
126 }
127
128 return ret;
129 }
130
131 case syscall::BF_MEM_OP_ALLOC_HUGE_IDX_VAL.get(): {
132 auto const ret{syscall_bf_mem_op_alloc_huge(mut_tls, mut_page_pool, mut_huge_pool)};
133 if (bsl::unlikely(ret != syscall::BF_STATUS_SUCCESS)) {
134 bsl::print<bsl::V>() << bsl::here();
135 return ret;
136 }
137
138 return ret;
139 }
140
141 default: {
142 break;
143 }
144 }
145
146 return report_syscall_unknown_unsupported(mut_tls);
147 }
