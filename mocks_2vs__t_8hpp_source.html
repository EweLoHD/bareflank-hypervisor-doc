131 [[nodiscard]] constexpr auto
132 allocate(
133 tls_t const &tls,
134 page_pool_t const &page_pool,
135 intrinsic_t const &intrinsic,
136 bsl::safe_u16 const &vmid,
137 bsl::safe_u16 const &vpid,
138 bsl::safe_u16 const &ppid) noexcept -> bsl::safe_u16
139 {
140 bsl::discard(tls);

145 bsl::expects(allocated_status_t::deallocated == m_allocated);

147 bsl::expects(vmid.is_valid_and_checked());
148 bsl::expects(vmid != syscall::BF_INVALID_ID);
149 bsl::expects(vpid.is_valid_and_checked());
150 bsl::expects(vpid != syscall::BF_INVALID_ID);
151 bsl::expects(ppid.is_valid_and_checked());
152 bsl::expects(ppid != syscall::BF_INVALID_ID);
153
154 m_assigned_vmid = ~vmid;
155 m_assigned_vpid = ~vpid;
156 m_assigned_ppid = ~ppid;
157 m_allocated = allocated_status_t::allocated;

395 [[nodiscard]] constexpr auto
396 read(tls_t const &tls, intrinsic_t const &intrinsic, syscall::bf_reg_t const reg)
397 const noexcept -> bsl::safe_umx
398 {
399 bsl::discard(intrinsic);
400
401 bsl::expects(allocated_status_t::allocated == m_allocated);
402 bsl::expects(tls.ppid == this->assigned_pp());
403
404 if (UNIT_TEST_VS_FAIL_READ == tls.test_ret) {
405 return bsl::safe_umx::failure();
406 }
407
408 if (reg == syscall::bf_reg_t::bf_reg_t_unsupported) {
409 return bsl::safe_umx::failure();
410 }
411
412 if (reg == syscall::bf_reg_t::bf_reg_t_invalid) {
413 return bsl::safe_umx::failure();
414 }
415
416 return m_regs.at(reg);
417 }

431 [[nodiscard]] constexpr auto

472 [[nodiscard]] constexpr auto
473 run(tls_t const &tls, intrinsic_t const &intrinsic, vmexit_log_t const &log) noexcept
474 -> bsl::safe_umx
475 {
476 bsl::discard(intrinsic);
477 bsl::discard(log);
478
479 bsl::expects(allocated_status_t::allocated == m_allocated);
480 bsl::expects(tls.ppid == this->assigned_pp());
481
482 if (tls.test_ret) {
483 if (bsl::safe_u64::magic_0() != tls.first_launch_succeeded) {
484 return bsl::safe_umx::failure();
485 }
486
487 bsl::touch();
488 }
489 else {
490 bsl::touch();
491 }
492
493 return {};
494 }
