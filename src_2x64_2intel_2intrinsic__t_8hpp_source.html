94 static constexpr void
95 tlb_flush(
96 bsl::safe_u64 const &addr, bsl::safe_u16 const &vpid = syscall::BF_INVALID_ID) noexcept
97 {
98 bsl::expects(addr.is_valid_and_checked());
99 bsl::expects(vpid.is_valid_and_checked());
100
101 if (syscall::BF_INVALID_ID == vpid) {
102 return intrinsic_invlpg(addr.get());
103 }
104
105 if (vpid.is_zero()) {
106 constexpr auto type{2_u64};
107 invept_descriptor_t const desc{{}, {}};
108 return intrinsic_invept(&desc, type.get());
109 }
110
111 if (addr.is_zero()) {
112 constexpr auto type{1_u64};
113 invept_descriptor_t const desc{vmrd64(VMCS_EPTP_INDEX).get(), {}};
114 return intrinsic_invept(&desc, type.get());
115 }
116
117 constexpr auto type{0_u64};
118 invvpid_descriptor_t const desc{vpid.get(), {}, {}, {}, addr.get()};
119 return intrinsic_invvpid(&desc, type.get());
120 }
