161 [[nodiscard]] constexpr auto
162 allocate(tls_t const &tls, bsl::safe_umx const &pages) noexcept -> bsl::span<page_4k_t>
163 {
164 lock_guard_t mut_lock{tls, m_lock};
165
166 bsl::expects(pages.is_valid_and_checked());
167 bsl::expects(pages.is_pos());
168
169 auto mut_buf{m_pool.subspan(bsl::to_idx(m_crsr), pages)};
170 if (bsl::unlikely(mut_buf.size() != pages)) {
171 bsl::error() << "huge pool out of memory\n" << bsl::here();
172 return {};
173 }
174
180
181 m_crsr = (m_crsr + pages).checked();
182 bsl::builtin_memset(mut_buf.data(), '\0', mut_buf.size_bytes());
183
184 bsl::ensures(m_crsr.is_valid_and_checked());
185 return mut_buf;
186 }
