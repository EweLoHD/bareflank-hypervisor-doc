144 if (bsl::unlikely(ret != syscall::BF_STATUS_SUCCESS)) {
145 bsl::print<bsl::V>() << bsl::here();
146 return ret;
147 }
148
149 return ret;
150 }
151
152 case syscall::BF_CALLBACK_OP_REGISTER_VMEXIT_IDX_VAL.get(): {
153 auto const ret{syscall_bf_callback_op_register_vmexit(mut_tls)};
154 if (bsl::unlikely(ret != syscall::BF_STATUS_SUCCESS)) {
155 bsl::print<bsl::V>() << bsl::here();
156 return ret;
157 }
158
159 return ret;
160 }
161

163 auto const ret{syscall_bf_callback_op_register_fail(mut_tls)};
164 if (bsl::unlikely(ret != syscall::BF_STATUS_SUCCESS)) {
165 bsl::print<bsl::V>() << bsl::here();
166 return ret;
167 }

169 return ret;
170 }
