460 [[nodiscard]] constexpr auto
461 dispatch_esr(tls_t &mut_tls, intrinsic_t &mut_intrinsic) noexcept -> bsl::errc_type
462 {
463 constexpr auto ec_shft{32_u64};
464
465 if (bsl::unlikely(bsl::safe_u64::magic_0() != mut_tls.mk_handling_esr)) {
466 bsl::alert() << "nested exception detected\n";
467 return return_failure(mut_tls);
468 }
469
470 mut_tls.mk_handling_esr = bsl::safe_u64::magic_1().get();
471
472 switch (mut_tls.esr_vector) {
473 case EXCEPTION_VECTOR_2.get(): {
474 dispatch_esr_nmi(mut_tls, mut_intrinsic);
475 return bsl::errc_success;
476 }
477
478 default: {
479 break;
480 }
481 }
482
483 auto const errc{(mut_tls.esr_error_code << ec_shft) | mut_tls.esr_vector};
484 auto const addr{bsl::to_u64(mut_tls.esr_pf_addr)};
485
486 if (send_exception_to_ext(mut_tls, mut_intrinsic, errc, addr)) {
487 return bsl::errc_success;
488 }
489
490 return return_failure(mut_tls);
491 }
