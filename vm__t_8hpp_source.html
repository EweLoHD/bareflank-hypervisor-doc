221 [[nodiscard]] constexpr auto
222 is_active(tls_t const &tls) const noexcept -> bsl::safe_u16
223 {
224 auto const online_pps{bsl::to_umx(tls.online_pps)};
225 bsl::expects(online_pps <= m_active.size());
226
227 for (bsl::safe_idx mut_i{}; mut_i < online_pps; ++mut_i) {
228 if (*m_active.at_if(mut_i)) {
229 return bsl::to_u16(mut_i);
230 }
231
232 bsl::touch();
233 }
234
235 return bsl::safe_u16::failure();
