373 [[nodiscard]] constexpr auto
374 initialize_bsp(
375 tls_t &mut_tls,
376 page_pool_t &mut_page_pool,
377 huge_pool_t &mut_huge_pool,
378 intrinsic_t &mut_intrinsic,
379 vm_pool_t &mut_vm_pool,
380 vp_pool_t &mut_vp_pool,
381 vs_pool_t &mut_vs_pool,
382 ext_pool_t &mut_ext_pool,
383 root_page_table_t &mut_system_rpt,
384 loader::mk_args_t &mut_args) noexcept -> bsl::errc_type
385 {
386 bsl::errc_type mut_ret{};
387
388 mut_page_pool.initialize(mut_args.page_pool);
389 mut_huge_pool.initialize(mut_args.huge_pool);
390
391 mut_ret = mut_system_rpt.initialize(mut_tls, mut_page_pool);
392 if (bsl::unlikely(!mut_ret)) {
393 bsl::print<bsl::V>() << bsl::here();
394 return bsl::errc_failure;
395 }
396
397 mut_system_rpt.add_tables(mut_tls, mut_args.rpt);
398
399 mut_vs_pool.initialize();
400 mut_vp_pool.initialize();
401 mut_vm_pool.initialize();
402
403 mut_ret = mut_ext_pool.initialize(
404 mut_tls, mut_page_pool, mut_huge_pool, mut_system_rpt, mut_args.ext_elf_files);
405 if (bsl::unlikely(!mut_ret)) {
406 bsl::print<bsl::V>() << bsl::here();
407 return bsl::errc_failure;
408 }
409
410 m_root_vmid = mut_vm_pool.allocate(mut_tls, mut_page_pool, mut_ext_pool);
411 if (bsl::unlikely(m_root_vmid.is_invalid())) {
412 bsl::print<bsl::V>() << bsl::here();
413 return bsl::errc_failure;
414 }
415
416 mut_vm_pool.set_active(mut_tls, m_root_vmid);
417
418 mut_ret = mut_ext_pool.start(mut_tls, mut_intrinsic);
419 if (bsl::unlikely(!mut_ret)) {
420 bsl::print<bsl::V>() << bsl::here();
421 return bsl::errc_failure;
422 }
423
424 if (bsl::unlikely(nullptr == mut_tls.ext_vmexit)) {
425 bsl::error() << "a vmexit handler was not registered\n" << bsl::here();
426 return bsl::errc_failure;
427 }
428
429 if (bsl::unlikely(nullptr == mut_tls.ext_fail)) {
430 bsl::error() << "a fail handler was not registered\n" << bsl::here();
431 return bsl::errc_failure;
432 }
433
434 m_ext_vmexit = mut_tls.ext_vmexit;
435 m_ext_fail = mut_tls.ext_fail;
436
437 bsl::ensures(m_root_vmid == syscall::BF_ROOT_VMID);
438 bsl::ensures(nullptr != m_ext_vmexit);
439 bsl::ensures(nullptr != m_ext_fail);
440
441 return bsl::errc_success;
442 }

486 [[nodiscard]] constexpr auto
487 process(
488 tls_t &mut_tls,
489 page_pool_t &mut_page_pool,
490 huge_pool_t &mut_huge_pool,
491 intrinsic_t &mut_intrinsic,
492 vm_pool_t &mut_vm_pool,
493 vp_pool_t &mut_vp_pool,
494 vs_pool_t &mut_vs_pool,
495 ext_pool_t &mut_ext_pool,
496 root_page_table_t &mut_system_rpt,
497 vmexit_log_t &mut_log,
498 loader::mk_args_t &mut_args) noexcept -> bsl::errc_type
499 {
500 bsl::errc_type mut_ret{};
501
506
507 bsl::expects(syscall::BF_INVALID_ID == mut_tls.active_vmid);
508 bsl::expects(syscall::BF_INVALID_ID == mut_tls.active_vpid);
509 bsl::expects(syscall::BF_INVALID_ID == mut_tls.active_vsid);
510
516
517 verify_mut_args(mut_args, mut_tls);
518
524
525 if (mut_args.ppid == syscall::BF_BS_PPID) {
526 print_logo();
527 }
528 else {
529 bsl::touch();
530 }
531
538
539 set_extension_sp(mut_tls);
540 set_extension_fail_sp(mut_tls);
541 set_extension_tp(mut_tls, mut_intrinsic);
542
547
548 if (mut_args.ppid == syscall::BF_BS_PPID) {
549 mut_ret = this->initialize_bsp(
550 mut_tls,
551 mut_page_pool,
552 mut_huge_pool,
553 mut_intrinsic,
554 mut_vm_pool,
555 mut_vp_pool,
556 mut_vs_pool,
557 mut_ext_pool,
558 mut_system_rpt,
559 mut_args);
560
561 if (bsl::unlikely(!mut_ret)) {
562 bsl::print<bsl::V>() << bsl::here();
563 return bsl::errc_failure;
564 }
565
566 bsl::touch();
567 }
568 else {
569 this->initialize_ap(mut_tls, mut_vm_pool);
570 }
571
576
577 mut_ret = mut_ext_pool.bootstrap(mut_tls, mut_intrinsic);
578 if (bsl::unlikely(!mut_ret)) {
579 bsl::print<bsl::V>() << bsl::here();
580 return bsl::errc_failure;
581 }
582
586
587 bsl::ensures(syscall::BF_INVALID_ID != mut_tls.ppid);
588 bsl::ensures(syscall::BF_INVALID_ID != mut_tls.active_vmid);
589 bsl::ensures(syscall::BF_INVALID_ID != mut_tls.active_vpid);
590 bsl::ensures(syscall::BF_INVALID_ID != mut_tls.active_vsid);
591
592 bsl::ensures(nullptr != mut_tls.active_rpt);
593 bsl::ensures(syscall::BF_INVALID_ID != mut_tls.active_extid);
594
598
599 return vmexit_loop(mut_tls, mut_intrinsic, mut_vs_pool, mut_log);
600 }
