131 [[nodiscard]] constexpr auto
132 allocate(
133 tls_t &mut_tls,
134 page_pool_t &mut_page_pool,
135 intrinsic_t &mut_intrinsic,
136 bsl::safe_u16 const &vmid,
137 bsl::safe_u16 const &vpid,
138 bsl::safe_u16 const &ppid) noexcept -> bsl::safe_u16
139 {
140 lock_guard_t mut_lock{mut_tls, m_lock};
141
142 for (auto &mut_vs : m_pool) {
143 if (mut_vs.is_deallocated()) {
144 return mut_vs.allocate(mut_tls, mut_page_pool, mut_intrinsic, vmid, vpid, ppid);
145 }
146
147 bsl::touch();
148 }
149
150 bsl::error() << "vs_pool_t out of vss\n" << bsl::here();
151 return bsl::safe_u16::failure();
152 }

334 vs_assigned_to_vm(bsl::safe_u16 const &vmid) const noexcept -> bsl::safe_u16
335 {
336 bsl::expects(vmid.is_valid_and_checked());
337 bsl::expects(vmid != syscall::BF_INVALID_ID);
338
339 for (auto const &vs : m_pool) {

550 constexpr void
551 tlb_flush(tls_t &mut_tls, intrinsic_t const &intrinsic, bsl::safe_u16 const &vmid) noexcept
552 {
553 bsl::expects(vmid.is_valid_and_checked());
554 bsl::expects(vmid != syscall::BF_INVALID_ID);
555
556 for (auto &mut_vs : m_pool) {
557 if (vmid != mut_vs.assigned_vm()) {
558 continue;
559 }
560
561 if (mut_tls.ppid != mut_vs.assigned_pp()) {
562 continue;
563 }
564
565 mut_vs.tlb_flush(mut_tls, intrinsic);
566 }
567 }
