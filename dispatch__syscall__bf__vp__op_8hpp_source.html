54 syscall_bf_vp_op_create_vp(
55 tls_t &mut_tls, vm_pool_t const &vm_pool, vp_pool_t &mut_vp_pool) noexcept
56 -> syscall::bf_status_t
57 {
58 auto const vmid{get_allocated_vmid(mut_tls.ext_reg1, vm_pool)};
59 if (bsl::unlikely(vmid.is_invalid())) {
60 bsl::print<bsl::V>() << bsl::here();
61 return syscall::BF_STATUS_INVALID_INPUT_REG1;
62 }
63
64 auto const vpid{mut_vp_pool.allocate(mut_tls, vmid)};
65 if (bsl::unlikely(vpid.is_invalid())) {
66 bsl::print<bsl::V>() << bsl::here();
67 return syscall::BF_STATUS_FAILURE_UNKNOWN;
68 }
69
70 mut_tls.ext_reg0 = bsl::merge_umx_with_u16(mut_tls.ext_reg0, vpid).get();
71 return syscall::BF_STATUS_SUCCESS;
72 }

114 [[nodiscard]] constexpr auto
115 dispatch_syscall_bf_vp_op(
116 tls_t &mut_tls,
117 vm_pool_t const &vm_pool,
118 vp_pool_t &mut_vp_pool,
119 vs_pool_t const &vs_pool) noexcept -> syscall::bf_status_t
120 {
121 if (bsl::unlikely(!verify_handle_for_current_ext(mut_tls))) {
122 bsl::print<bsl::V>() << bsl::here();
123 return syscall::BF_STATUS_FAILURE_INVALID_HANDLE;
124 }
125
126 if (bsl::unlikely(!is_the_active_ext_the_vmexit_ext(mut_tls))) {
127 bsl::print<bsl::V>() << bsl::here();
128 return syscall::BF_STATUS_INVALID_PERM_DENIED;
129 }
130
131 switch (syscall::bf_syscall_index(mut_tls.ext_syscall).get()) {
132 case syscall::BF_VP_OP_CREATE_VP_IDX_VAL.get(): {
133 auto const ret{syscall_bf_vp_op_create_vp(mut_tls, vm_pool, mut_vp_pool)};
134 if (bsl::unlikely(ret != syscall::BF_STATUS_SUCCESS)) {
135 bsl::print<bsl::V>() << bsl::here();
136 return ret;
137 }
138
139 return ret;
140 }
141
142 case syscall::BF_VP_OP_DESTROY_VP_IDX_VAL.get(): {
143 auto const ret{syscall_bf_vp_op_destroy_vp(mut_tls, mut_vp_pool, vs_pool)};
144 if (bsl::unlikely(ret != syscall::BF_STATUS_SUCCESS)) {
145 bsl::print<bsl::V>() << bsl::here();
146 return ret;
147 }
148
149 return ret;
150 }
151
152 default: {
153 break;
154 }
155 }
156
157 return report_syscall_unknown_unsupported(mut_tls);
158 }
