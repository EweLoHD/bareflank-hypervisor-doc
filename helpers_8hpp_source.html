314 constexpr void
315 init_file(file_t &mut_file) noexcept
316 {
317 mut_file.ehdr.e_ident[bfelf_ei_mag0] = bfelf_elfmag0;
318 mut_file.ehdr.e_ident[bfelf_ei_mag1] = bfelf_elfmag1;
319 mut_file.ehdr.e_ident[bfelf_ei_mag2] = bfelf_elfmag2;
320 mut_file.ehdr.e_ident[bfelf_ei_mag3] = bfelf_elfmag3;
321 mut_file.ehdr.e_ident[bfelf_ei_class] = bfelf_elfclass64;
322 mut_file.ehdr.e_ident[bfelf_ei_osabi] = bfelf_elfosabi_sysv;
323 mut_file.ehdr.e_type = bfelf_et_exec;
324 mut_file.ehdr.e_phdr = phdrtbl_offset(mut_file);
325 mut_file.ehdr.e_phnum = bsl::to_u16(mut_file.phdrtbl.size()).get();
326 mut_file.ehdr.e_shdr = shdrtbl_offset(mut_file);
327 mut_file.ehdr.e_shnum = bsl::to_u16(mut_file.shdrtbl.size()).get();
328
329 for (auto &mut_phdr : mut_file.phdrtbl) {
330 mut_phdr.p_offset = segment_offset(mut_file);
331 mut_phdr.p_filesz = mut_file.segment.size().get();
332 mut_phdr.p_memsz = mut_file.segment.size().get();
333 }
334
335 for (auto &mut_shdr : mut_file.shdrtbl) {
336 mut_shdr.sh_offset = section_offset(mut_file);
337 mut_shdr.sh_size = bsl::to_u32(mut_file.section.size()).get();
338 }
339
340 mut_file.phdrtbl.front().p_type = bfelf_pt_load;
341 }
