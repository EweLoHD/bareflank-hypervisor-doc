66 [[nodiscard]] constexpr auto
67 dispatch_syscall_bf_debug_op(
68 tls_t &mut_tls,
69 page_pool_t const &page_pool,
70 huge_pool_t const &huge_pool,
71 intrinsic_t const &intrinsic,
72 vm_pool_t const &vm_pool,
73 vp_pool_t const &vp_pool,
74 vs_pool_t const &vs_pool,
75 ext_pool_t const &ext_pool,
76 vmexit_log_t const &log) noexcept -> syscall::bf_status_t
77 {
78 switch (syscall::bf_syscall_index(mut_tls.ext_syscall).get()) {
79 case syscall::BF_DEBUG_OP_OUT_IDX_VAL.get(): {
80 bsl::print() << bsl::hex(mut_tls.ext_reg0) //--
81 << " " //--
82 << bsl::hex(mut_tls.ext_reg1) //--
83 << bsl::endl; //--
84 return syscall::BF_STATUS_SUCCESS;
85 }
86
87 case syscall::BF_DEBUG_OP_DUMP_VM_IDX_VAL.get(): {
88 auto const vmid{get_vmid(mut_tls.ext_reg0)};
89 if (bsl::unlikely(vmid.is_invalid())) {
90 bsl::print<bsl::V>() << bsl::here();
91 return syscall::BF_STATUS_INVALID_INPUT_REG0;
92 }
93
94 vm_pool.dump(mut_tls, vmid);
95 return syscall::BF_STATUS_SUCCESS;
96 }
97
98 case syscall::BF_DEBUG_OP_DUMP_VP_IDX_VAL.get(): {
99 auto const vpid{get_vpid(mut_tls.ext_reg0)};
100 if (bsl::unlikely(vpid.is_invalid())) {
101 bsl::print<bsl::V>() << bsl::here();
102 return syscall::BF_STATUS_INVALID_INPUT_REG0;
103 }
104
105 vp_pool.dump(vpid);
106 return syscall::BF_STATUS_SUCCESS;
107 }
108
109 case syscall::BF_DEBUG_OP_DUMP_VS_IDX_VAL.get(): {
110 auto const vsid{get_vsid(mut_tls.ext_reg0)};
111 if (bsl::unlikely(vsid.is_invalid())) {
112 bsl::print<bsl::V>() << bsl::here();
113 return syscall::BF_STATUS_INVALID_INPUT_REG0;
114 }
115
116 vs_pool.dump(mut_tls, intrinsic, vsid);
117 return syscall::BF_STATUS_SUCCESS;
118 }
119
120 case syscall::BF_DEBUG_OP_DUMP_VMEXIT_LOG_IDX_VAL.get(): {
121 auto const ppid{get_ppid(mut_tls, mut_tls.ext_reg0)};
122 if (bsl::unlikely(ppid.is_invalid())) {
123 bsl::print<bsl::V>() << bsl::here();
124 return syscall::BF_STATUS_INVALID_INPUT_REG0;
125 }
126
127 log.dump(ppid);
128 return syscall::BF_STATUS_SUCCESS;
129 }
130
131 case syscall::BF_DEBUG_OP_WRITE_C_IDX_VAL.get(): {
132 bsl::print() << static_cast<bsl::char_type>(bsl::to_u8(mut_tls.ext_reg0).get());
133 return syscall::BF_STATUS_SUCCESS;
134 }
135
136 case syscall::BF_DEBUG_OP_WRITE_STR_IDX_VAL.get(): {
137 // NOLINTNEXTLINE(cppcoreguidelines-pro-type-reinterpret-cast)
138 auto const *const str{reinterpret_cast<bsl::cstr_type>(mut_tls.ext_reg0)};
139 if (bsl::unlikely(nullptr == str)) {
140 return syscall::BF_STATUS_INVALID_INPUT_REG0;
141 }
142
158
159 bsl::print() << bsl::string_view{str, bsl::to_umx(mut_tls.ext_reg1)};
160 return syscall::BF_STATUS_SUCCESS;
161 }
162
163 case syscall::BF_DEBUG_OP_DUMP_EXT_IDX_VAL.get(): {
164 auto const extid{get_extid(mut_tls.ext_reg0)};
165 if (bsl::unlikely(extid.is_invalid())) {
166 bsl::print<bsl::V>() << bsl::here();
167 return syscall::BF_STATUS_INVALID_INPUT_REG0;
168 }
169
170 ext_pool.dump(mut_tls, extid);
171 return syscall::BF_STATUS_SUCCESS;
172 }
173
174 case syscall::BF_DEBUG_OP_DUMP_PAGE_POOL_IDX_VAL.get(): {
175 page_pool.dump(mut_tls);
176 return syscall::BF_STATUS_SUCCESS;
177 }
178
179 case syscall::BF_DEBUG_OP_DUMP_HUGE_POOL_IDX_VAL.get(): {
180 huge_pool.dump(mut_tls);
181 return syscall::BF_STATUS_SUCCESS;
182 }
183
184 default: {
185 break;
186 }
187 }
188
189 return report_syscall_unknown_unsupported(mut_tls);
190 }
