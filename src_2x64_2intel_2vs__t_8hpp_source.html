656 [[nodiscard]] constexpr auto
657 allocate(
658 tls_t &mut_tls,
659 page_pool_t &mut_page_pool,
660 intrinsic_t &mut_intrinsic,
661 bsl::safe_u16 const &vmid,
662 bsl::safe_u16 const &vpid,
663 bsl::safe_u16 const &ppid) noexcept -> bsl::safe_u16
664 {
665 bsl::expects(this->id() != syscall::BF_INVALID_ID);
666 bsl::expects(allocated_status_t::deallocated == m_allocated);
667
668 bsl::expects(vmid.is_valid_and_checked());
669 bsl::expects(vmid != syscall::BF_INVALID_ID);
670 bsl::expects(vpid.is_valid_and_checked());
671 bsl::expects(vpid != syscall::BF_INVALID_ID);
672 bsl::expects(ppid.is_valid_and_checked());
673 bsl::expects(ppid != syscall::BF_INVALID_ID);
674
675 m_vmcs = mut_page_pool.template allocate<vmcs_t>(mut_tls);
676 if (bsl::unlikely(nullptr == m_vmcs)) {
677 bsl::print<bsl::V>() << bsl::here();
678 return bsl::safe_u16::failure();
679 }
680
681 m_vmcs_phys = mut_page_pool.virt_to_phys(m_vmcs);
682 bsl::expects(m_vmcs_phys.is_valid_and_checked());
683
684 m_assigned_vmid = ~vmid;
685 m_assigned_vpid = ~vpid;
686 m_assigned_ppid = ~ppid;
687 m_allocated = allocated_status_t::allocated;
688
689 this->init_vmcs(mut_tls, mut_intrinsic);
690 return this->id();
691 }
