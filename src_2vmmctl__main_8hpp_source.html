86 [[nodiscard]] static constexpr auto
87 start_vmm(bsl::arguments &mut_args, ioctl_t &mut_ioctl) noexcept -> bsl::errc_type
88 {
89 loader::start_vmm_args_t mut_start_args{IOCTL_VERSION.get(), {}, {}, {}, {}};
90
91 auto const mk_filename{mut_args.front<bsl::string_view>()};
92 if (mk_filename.empty()) {
93 bsl::error() << "the microkernel's path is either missing or empty\n";
94 help();
95 return bsl::errc_failure;
96 }
97
98 ifmap_t const mk_map{mk_filename};
99 if (bsl::unlikely(mk_map.empty())) {
100 help();
101 return bsl::errc_failure;
102 }
103
104 mut_start_args.mk_elf_file = mk_map.view();
105
106 ++mut_args;
107 auto mut_ext_filename{mut_args.front<bsl::string_view>()};
108 if (mut_ext_filename.empty()) {
109 bsl::error() << "the extension's path is either missing or empty\n";
110 help();
111 return bsl::errc_failure;
112 }
113
114 bsl::array<ifmap_t, HYPERVISOR_MAX_EXTENSIONS.get()> mut_ext_maps{};
115 for (bsl::safe_idx mut_i{}; mut_i < HYPERVISOR_MAX_EXTENSIONS; ++mut_i) {
116 auto *const pmut_ext_map{mut_ext_maps.at_if(mut_i)};
117 auto *const pmut_ext_elf{mut_start_args.ext_elf_files.at_if(mut_i)};
118
119 *pmut_ext_map = ifmap_t{mut_ext_filename};
120 if (bsl::unlikely(pmut_ext_map->empty())) {
121 help();
122 return bsl::errc_failure;
123 }
124
125 *pmut_ext_elf = pmut_ext_map->view();
126
127 ++mut_args;
128 mut_ext_filename = mut_args.front<bsl::string_view>();
129 if (mut_ext_filename.empty()) {
130 break;
131 }
132
133 bsl::touch();
134 }
135
136 auto const ret{mut_ioctl.write(loader::START_VMM, &mut_start_args)};
137 if (bsl::unlikely(ret.is_neg())) {
138 bsl::error() << "vmmctl failed. check kernel logs details\n";
139 return bsl::errc_failure;
140 }
141
142 return bsl::errc_success;
143 }

177 [[nodiscard]] static constexpr auto
178 dump_vmm(ioctl_t &mut_ioctl) noexcept -> bsl::errc_type
179 {
180 loader::dump_vmm_args_t mut_dump_args{IOCTL_VERSION.get(), {}};
181
182 auto const ret{mut_ioctl.read_write(loader::DUMP_VMM, &mut_dump_args)};
183 if (bsl::unlikely(ret.is_neg())) {
184 bsl::error() << "vmmctl failed. check kernel logs details\n";
185 return bsl::errc_failure;
186 }
187
188 bsl::safe_idx mut_epos{mut_dump_args.debug_ring.epos};
189 bsl::safe_idx mut_spos{mut_dump_args.debug_ring.spos};
190
191 if (bsl::unlikely(mut_epos >= mut_dump_args.debug_ring.buf.size())) {
192 bsl::error() << "kernel returned an invalid debug ring\n";
193 return bsl::errc_failure;
194 }
195
196 if (bsl::unlikely(mut_spos >= mut_dump_args.debug_ring.buf.size())) {
197 bsl::error() << "kernel returned an invalid debug ring\n";
198 return bsl::errc_failure;
199 }
200
201 if (mut_spos == mut_epos) {
202 bsl::alert() << "no debug data to dump\n";
203 return bsl::errc_success;
204 }
205
206 while (mut_spos != mut_epos) {
207 if (mut_spos >= mut_dump_args.debug_ring.buf.size()) {
208 mut_spos = {};
209 }
210 else {
211 bsl::touch();
212 }
213
214 bsl::print() << *mut_dump_args.debug_ring.buf.at_if(mut_spos.get());
215 ++mut_spos;
216 }
217
218 bsl::print() << bsl::endl;
219 return bsl::errc_success;
220 }

236 [[nodiscard]] constexpr auto
237 process_cmd(bsl::arguments &mut_args, ioctl_t &mut_ioctl) noexcept -> bsl::errc_type
238 {
239 auto const cmd{mut_args.front<bsl::string_view>()};
240 ++mut_args;
241
242 if (cmd == "start") {
243 return this->start_vmm(mut_args, mut_ioctl);
244 }
245
246 if (cmd == "stop") {
247 return this->stop_vmm(mut_ioctl);
248 }
249
250 if (cmd == "dump") {
251 return this->dump_vmm(mut_ioctl);
252 }
253
254 if (cmd.empty()) {
255 bsl::error() << "missing command\n";
256 }
257 else {
258 bsl::error() << "invalid command: \"" << cmd << "\"\n";
259 }
260
261 help();
262 return bsl::errc_failure;
263 }
